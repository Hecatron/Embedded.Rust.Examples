struct ConsoleWriter {
    out: String,
}

// Default values
impl Default for ConsoleWriter {
    fn default() -> ConsoleWriter {
        ConsoleWriter {
            out: String::new(),
        }
    }
}

// Members
impl ConsoleWriter {
    pub fn append<I>(&mut self, val: I) where I: AppendString {
        self.push_str(&val.to_string())
    }
}

    fn append(self);
}
